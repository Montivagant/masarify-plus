class SubscriptionSuggestCard extends StatefulWidget {
  const SubscriptionSuggestCard({

    required this.categoryName,

        child: Padding(
          padding: const EdgeInsets.all(AppSizes.md),

              Icon(AppIcons.recurring,
                  color: context.appTheme.transferColor,
                  size: AppSizes.iconMd),
              const SizedBox(width: AppSizes.sm),
              Expanded(

                  style: Theme.of(context).textTheme.bodySmall,
                ),
              ),
              const SizedBox(width: AppSizes.xs),
              TextButton(
                onPressed: () => setState(() => _dismissed = true),
                child: Text(context.l10n.common_dismiss),

              FilledButton(
                onPressed: () {
                  context.push(AppRoutes.recurringAdd);

            ],

}

4. Add the handler method:
```dart
void _onTitleChanged() {
  _debounceTimer?.cancel();
  _debounceTimer = Timer(AppDurations.categorySuggestionDebounce, () async {
    final text = _titleController.text.trim();
    if (text.length < 3 || _categoryId != null) return; // Already selected or too short
    final cats = ref.read(categoriesProvider).valueOrNull ?? [];
    final compatible = cats.where((c) => c.type == _type || c.type == 'both').toList();
    final service = ref.read(categorizationLearningServiceProvider);
    final suggestion = await service.suggestFromText(text, compatible);
    if (mounted && suggestion != null && _categoryId == null) {
      setState(() => _suggestedCategory = suggestion);
    }
  });
}
```
