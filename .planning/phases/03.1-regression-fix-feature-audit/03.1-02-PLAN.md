```dart
/// Save all unsaved, valid drafts in a single batch operation (D-16).
Future<void> _saveAll(BuildContext ctx) async {
  if (_saving) return;
  setState(() => _saving = true);

  final l10n = ctx.l10n;
  final nav = GoRouter.of(ctx);
  int savedCount = 0;
  String? errorMsg;

  for (int i = 0; i < _editableDrafts.length; i++) {
    if (_savedFlags[i]) continue; // Skip already saved
    final draft = _editableDrafts[i];

    // Validate ŌĆö skip invalid drafts silently
    if (draft.amountPiastres <= 0) continue;
    if (draft.categoryId == null && !_isCashType(draft.type)) continue;
    if (draft.walletId == null) continue;

    try {
      await _saveDraft(draft);
      _savedFlags[i] = true;
      savedCount++;
    } catch (_) {
      errorMsg = l10n.common_error_generic;
      break;
    }
  }

  if (!mounted) return;

  if (errorMsg != null) {
    setState(() => _saving = false);
    SnackHelper.showError(context, errorMsg);
    return;
  }

  if (savedCount > 0) {
    HapticFeedback.heavyImpact();
    SnackHelper.showSuccess(context, l10n.voice_confirm_all_saved);
    nav.pop();
  } else {
    setState(() => _saving = false);
    SnackHelper.showInfo(
      context,
      l10n.error_amount_zero, // No valid drafts to save
    );
  }
}
```
