```dart
import 'package:flutter/material.dart';

import '../../../app/theme/app_colors.dart';
import '../../../core/constants/app_sizes.dart';
import '../../../core/utils/category_icon_mapper.dart';
import '../../../core/utils/color_utils.dart';
import '../glass/glass_card.dart';

/// A category icon rendered inside a colored glass circle.
///
/// Uses [GlassTier.inset] (sigma 8) with the category's color as a tint.
/// The icon itself is always white (Phosphor icon on colored glass).
/// Per D-01: consistent with the app's 3-tier glass system.
class GlassCategoryIcon extends StatelessWidget {
  const GlassCategoryIcon({
    super.key,
    required this.iconName,
    required this.colorHex,
    this.size = AppSizes.categoryChipSize,
    this.isSelected = false,
  });

  /// Maps to a Phosphor icon via [CategoryIconMapper.fromName].
  final String iconName;

  /// Category color hex string (e.g. '#E57373'). Used as tint at alpha 0.35.
  final String colorHex;

  /// Outer circle diameter. Defaults to [AppSizes.categoryChipSize] (52px).
  final double size;

  /// When true, adds a primary-colored border and primaryContainer fill.
  final bool isSelected;

  @override
  Widget build(BuildContext context) {
    final color = ColorUtils.fromHex(colorHex);
    final icon = CategoryIconMapper.fromName(iconName);
    final cs = Theme.of(context).colorScheme;

    Widget circle = Semantics(
      label: iconName,
      child: GlassCard(
        tier: GlassTier.inset,
        padding: EdgeInsets.zero,
        showBorder: true,
        tintColor: color.withValues(alpha: 0.35),
        borderRadius: BorderRadius.circular(size / 2),
        child: SizedBox(
          width: size,
          height: size,
          child: Center(
            child: Icon(
              icon,
              color: AppColors.white,
              size: size * 0.5,
            ),
          ),
        ),
      ),
    );

    if (isSelected) {
      circle = Container(
        decoration: BoxDecoration(
          color: cs.primaryContainer,
          borderRadius: BorderRadius.circular(size / 2),
          border: Border.all(
            color: cs.primary,
            width: AppSizes.borderWidthFocus,
          ),
        ),
        child: circle,
      );
    }

    return circle;
  }
}
```

**Create lib/shared/widgets/pickers/category_picker_sheet.dart:**

This is the unified picker per D-02. It has:
- DraggableScrollableSheet with sizes from AppSizes (initial 0.6, min 0.4, max 0.85)
- Glass card background using GlassTier.background
- Drag handle + "Select Category" title (headlineSmall, 20sp)
- Search bar with AppIcons.search prefix, borderRadiusFull
- "Most Used" section: top 5 by usageCount DESC where usageCount > 0, with star icon in accent color. Only shown when NOT searching.
- "All Categories" section: grouped by type (Expense header, then Income header). 4-column grid.
- Grid tile: GlassCategoryIcon (52px) + 4px gap + category name (bodySmall, 12sp, centered, maxLines 1, ellipsis)
- Search filters by name, nameAr, and iconName (case-insensitive). When searching, flat list replaces sections.
- Empty search: "No categories found" centered text

```dart
import 'package:flutter/material.dart';

import '../../../core/constants/app_sizes.dart';
import '../../../core/utils/context_extensions.dart'; // for context.l10n, context.colors, context.textStyles, context.languageCode
import '../../../domain/entities/category_entity.dart';
import '../category/glass_category_icon.dart';
import '../navigation/drag_handle.dart';

/// Unified category picker replacing 3 inline implementations.
/// Per D-02: DraggableScrollableSheet with "Most Used" + "All Categories" grouped by type.
class CategoryPickerSheet extends StatefulWidget {
  const CategoryPickerSheet({
    super.key,
    required this.categories,
    required this.onSelected,
    this.selectedId,
  });

  /// Full non-archived category list.
  final List<CategoryEntity> categories;

  /// Currently selected category ID (for highlight).
  final int? selectedId;

  /// Fires when user taps a category.
  final ValueChanged<int> onSelected;

  @override
  State<CategoryPickerSheet> createState() => _CategoryPickerSheetState();
}

class _CategoryPickerSheetState extends State<CategoryPickerSheet> {
  final _searchController = TextEditingController();
  String _query = '';

  @override
  void dispose() {
    _searchController.dispose();
    super.dispose();
  }

  void _onSearch() {
    setState(() => _query = _searchController.text.trim().toLowerCase());
  }

  List<CategoryEntity> get _filtered {
    if (_query.isEmpty) return widget.categories;
    return widget.categories.where((c) {
      return c.name.toLowerCase().contains(_query) ||
          c.nameAr.contains(_query) ||
          c.iconName.toLowerCase().contains(_query);
    }).toList();
  }

  List<CategoryEntity> get _mostUsed {
    final sorted = [...widget.categories]
      ..sort((a, b) => b.usageCount.compareTo(a.usageCount));
    return sorted.where((c) => c.usageCount > 0).take(5).toList();
  }

  @override
  void initState() {
    super.initState();
    _searchController.addListener(_onSearch);
  }

  @override
  Widget build(BuildContext context) {
    final l10n = context.l10n;
    final cs = context.colors;
    final styles = context.textStyles;
    final locale = context.languageCode;
    final isSearching = _query.isNotEmpty;
    final filtered = _filtered;
    final mostUsed = _mostUsed;

    return DraggableScrollableSheet(
      initialChildSize: AppSizes.sheetInitialSize,
      minChildSize: AppSizes.sheetSmallInitialSize,
      maxChildSize: AppSizes.sheetMaxSize,
      expand: false,
      builder: (ctx, scrollController) => Column(
        children: [
          const DragHandle(),
          // Title
          Padding(
            padding: const EdgeInsets.symmetric(horizontal: AppSizes.md),
            child: Align(
              alignment: AlignmentDirectional.centerStart,
              child: Text(
                l10n.transaction_category_picker,
                style: styles.headlineSmall,
              ),
            ),
          ),
          const SizedBox(height: AppSizes.sm),
          // Search bar
          Padding(
            padding: const EdgeInsets.symmetric(horizontal: AppSizes.md),
            child: TextField(
              controller: _searchController,
              decoration: InputDecoration(
                hintText: l10n.category_search_hint,
                prefixIcon: Icon(AppIcons.search, size: AppSizes.iconSm),
                suffixIcon: _query.isNotEmpty
                    ? IconButton(
                        icon: const Icon(Icons.close),
                        onPressed: () {
                          _searchController.clear();
                        },
                      )
                    : null,
                isDense: true,
                contentPadding: const EdgeInsets.symmetric(
                  horizontal: AppSizes.md,
                  vertical: AppSizes.sm,
                ),
                border: OutlineInputBorder(
                  borderRadius: BorderRadius.circular(AppSizes.borderRadiusFull),
                ),
              ),
            ),
          ),
          const SizedBox(height: AppSizes.sm),
          // Content
          Expanded(
            child: isSearching
                ? _buildSearchResults(filtered, locale, cs, styles, scrollController)
                : _buildSections(filtered, mostUsed, locale, cs, styles, scrollController),
          ),
        ],
      ),
    );
  }

  Widget _buildSearchResults(
    List<CategoryEntity> results,
    String locale,
    ColorScheme cs,
    dynamic styles,
    ScrollController scrollController,
  ) {
    if (results.isEmpty) {
      return Center(
        child: Text(
          context.l10n.category_search_empty,
          style: context.textStyles.bodySmall?.copyWith(color: cs.outline),
        ),
      );
    }
    return GridView.builder(
      controller: scrollController,
      padding: const EdgeInsetsDirectional.fromSTEB(
        AppSizes.md, AppSizes.xs, AppSizes.md, AppSizes.bottomScrollPadding,
      ),
      gridDelegate: const SliverGridDelegateWithFixedCrossAxisCount(
        crossAxisCount: 4,
        childAspectRatio: 0.85,
        crossAxisSpacing: AppSizes.xs,
        mainAxisSpacing: AppSizes.xs,
      ),
      itemCount: results.length,
      itemBuilder: (_, i) => _buildTile(results[i], locale, cs),
    );
  }

  Widget _buildSections(
    List<CategoryEntity> all,
    List<CategoryEntity> mostUsed,
    String locale,
    ColorScheme cs,
    dynamic styles,
    ScrollController scrollController,
  ) {
    final expenses = all.where((c) => c.type == 'expense' || c.type == 'both').toList();
    final incomes = all.where((c) => c.type == 'income' || c.type == 'both').toList();

    return ListView(
      controller: scrollController,
      padding: const EdgeInsetsDirectional.fromSTEB(
        AppSizes.md, 0, AppSizes.md, AppSizes.bottomScrollPadding,
      ),
      children: [
        // Most Used section (D-13)
        if (mostUsed.isNotEmpty) ...[
          _SectionHeader(
            label: context.l10n.category_most_used,
            icon: Icons.star_rounded,
            iconColor: cs.primary,
          ),
          const SizedBox(height: AppSizes.sm),
          _buildGrid(mostUsed, locale, cs),
          const SizedBox(height: AppSizes.lg),
        ],
        // Expense section
        if (expenses.isNotEmpty) ...[
          _SectionHeader(label: context.l10n.categories_expense),
          const SizedBox(height: AppSizes.sm),
          _buildGrid(expenses, locale, cs),
          const SizedBox(height: AppSizes.lg),
        ],
        // Income section
        if (incomes.isNotEmpty) ...[
          _SectionHeader(label: context.l10n.categories_income),
          const SizedBox(height: AppSizes.sm),
          _buildGrid(incomes, locale, cs),
        ],
      ],
    );
  }

  Widget _buildGrid(List<CategoryEntity> cats, String locale, ColorScheme cs) {
    return GridView.builder(
      shrinkWrap: true,
      physics: const NeverScrollableScrollPhysics(),
      gridDelegate: const SliverGridDelegateWithFixedCrossAxisCount(
        crossAxisCount: 4,
        childAspectRatio: 0.85,
        crossAxisSpacing: AppSizes.xs,
        mainAxisSpacing: AppSizes.xs,
      ),
      itemCount: cats.length,
      itemBuilder: (_, i) => _buildTile(cats[i], locale, cs),
    );
  }

  Widget _buildTile(CategoryEntity cat, String locale, ColorScheme cs) {
    final isSelected = cat.id == widget.selectedId;
    return GestureDetector(
      onTap: () => widget.onSelected(cat.id),
      child: Column(
        mainAxisSize: MainAxisSize.min,
        children: [
          GlassCategoryIcon(
            iconName: cat.iconName,
            colorHex: cat.colorHex,
            size: AppSizes.categoryChipSize,
            isSelected: isSelected,
          ),
          const SizedBox(height: AppSizes.xs),
          Text(
            cat.displayName(locale),
            style: context.textStyles.bodySmall,
            textAlign: TextAlign.center,
            maxLines: 1,
            overflow: TextOverflow.ellipsis,
          ),
        ],
      ),
    );
  }
}

class _SectionHeader extends StatelessWidget {
  const _SectionHeader({required this.label, this.icon, this.iconColor});
  final String label;
  final IconData? icon;
  final Color? iconColor;

  @override
  Widget build(BuildContext context) {
    return Row(
      children: [
        if (icon != null) ...[
          Icon(icon, size: AppSizes.iconSm, color: iconColor),
          const SizedBox(width: AppSizes.xs),
        ],
        Text(
          label,
          style: context.textStyles.titleMedium?.copyWith(
            fontWeight: FontWeight.w600,
          ),
        ),
      ],
    );
  }
}
```
